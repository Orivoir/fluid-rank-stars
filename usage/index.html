.js 

                        
document.addEventListener('DOMContentLoaded' , () => {

    fluidRankInit({
        // optional
        background: {
            // default : "silver"
            // active: "goldenrod"
        } ,
        flex: true , // vertical align stars
        autoResponsive: true ,
        // size: 50 (default 42 (px unity) ) ,

        // Optional but warning after valid rank (click on the rank validate) 
        onValid: function( data ) {

            console.log( 'you have valid an rank , data : '  , data );
        }
    }) ;

} ) ;
